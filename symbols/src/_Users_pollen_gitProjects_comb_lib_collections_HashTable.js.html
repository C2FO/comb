  5 var hashFunction = function (key) {
  6     if (typeof key == "string") {
  7         return key;
  8     } else if (typeof key == "object") {
  9         return  key.hashCode ? key.hashCode() : "" + key;
 10     } else {
 11         return "" + key;
 12     }
