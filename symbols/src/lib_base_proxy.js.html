  3 var comb = exports;
  4 var handlerMaker = function (obj) {
  5     return {
  6         getOwnPropertyDescriptor:function (name) {
  7             var desc = Object.getOwnPropertyDescriptor(obj, name);
  8             // a trapping proxy's properties must always be configurable
  9             if (desc !== undefined) {
 10                 desc.configurable = true;
 11             }
 12             return desc;
 13         },
 14         getPropertyDescriptor:function (name) {
 15             var desc = Object.getPropertyDescriptor(obj, name); // not in ES5
 16             // a trapping proxy's properties must always be configurable
 17             if (desc !== undefined) {
 18                 desc.configurable = true;
 19             }
 20             return desc;
 21         },
 22         getOwnPropertyNames:function () {
 23             return Object.getOwnPropertyNames(obj);
 24         },
 25         getPropertyNames:function () {
 26             return Object.getPropertyNames(obj);                // not in ES5
 27         },
 28         defineProperty:function (name, desc) {
 29             Object.defineProperty(obj, name, desc);
 30         },
 31         delete:function (name) {
 32             return delete obj[name];
 33         },
 34         fix:function () {
 35             if (Object.isFrozen(obj)) {
 36                 var result = {};
 37                 Object.getOwnPropertyNames(obj).forEach(function (name) {
 38                     result[name] = Object.getOwnPropertyDescriptor(obj, name);
 39                 });
 40                 return result;
 41             }
 42             // As long as obj is not frozen, the proxy won't allow itself to be fixed
 43             return undefined; // will cause a TypeError to be thrown
 44         },
 45 
 46         has:function (name) {
 47             return name in obj;
 48         },
 49         hasOwn:function (name) {
 50             return ({}).hasOwnProperty.call(obj, name);
 51         },
 52         get:function (receiver, name) {
 53             return obj[name];
 54         },
 55         set:function (receiver, name, val) {
 56             obj[name] = val;
 57             return true;
 58         }, // bad behavior when set fails in non-strict mode
 59         enumerate:function () {
 60             var result = [];
 61             for (var name in obj) {
 62                 result.push(name);
 63             }
 64             return result;
 65         },
 66         keys:function () {
 67             return Object.keys(obj);
 68         }
 69 
 70     };
 71 };

 73 var noSuchMethodHandler = function (obj, handler) {
 74     return {
 75         get:function (receiver, name) {
 76             return obj[name] ? obj[name] : handler.call(obj, name);
 77         }
 78     }
 79 };

 81 /**
 82  * Creates a proxy for an object.
 83  * @param obj object to proxy
 84  * @param {Object} opts object with methods to define on the handler.
 85  * @memberOf comb
 86  */
 87 comb.handlerProxy = function (obj, opts, proto) {
 88     opts = opts || {};
 89     if (misc.isUndefined(proto)) {
 90         return  Proxy.create(merge(handlerMaker(obj), opts));
 91     } else {
 92         return  Proxy.create(merge(handlerMaker(obj), opts), object.isHash(proto) ? proto : proto.prototype);
 93     }
 94 };

 96 /**
 97  * Creates a method missing proxy for an object.
 98  * <b>NOTE:</b> This method does not gurantee that the property will be used as a function call.
 99  *
100  * @example
101  *
102  *  var x = {hello:function () {return "hello"}, world:"world"};
103  *  var xHandler = comb.methodMissing(x, function (m) {
104  *              //you can do more interesting stuff in here!
105  *               return function () {
106  *                   return [m].concat(comb.argsToArray(arguments));
107  *               }
108  *   });
109  *  xHandler.hello(); //=> "hello"
110  *  xHandler.world //=> "world"
111  *  xHandler.someMethod("hello", "world"); //=> [ 'someMethod', 'hello', 'world' ]
112  *
113  * @param {Object} obj object to wrap with a method missing proxy
114  * @param {Function} handler handle to call when a property is missing
115  * @param {Object} opts prototype to assign to the proxy
116  * @memberOf comb
117  * @returns {Proxy} a proxy
118  */
119 comb.methodMissing = function (obj, handler, proto) {
120     proto = proto || {};
121     return  Proxy.create(merge(handlerMaker(obj), noSuchMethodHandler(obj, handler)), object.isHash(proto) ? proto : proto.prototype);
122 };

136 /**
137  * Creates a function proxy for an object.
138  *
139  * @example
140  *
141  * //create an object that can use properties or as a function through the new operator
142  * var MyObject = comb.define(null, {
143  *     instance : {
144  *         hello : "hello",
145  *         constructor : function(){
146  *             this.args = comb.argsToArray(arguments);
147  *         }
148  *     }
149  * });
150  *
151  * //NOTE: this will not work properly for native objects like Date.
152  * var createNewMyObject = function(){
153  *    try {
154  *      p = new MyObject();
155  *     } catch (ignore) {
156  *          //ignore the error because its probably from missing arguments
157  *     }
158  *     //Now lets take care of arguments supplied!!!
159  *     MyObject.apply(p, comb.argsToArray(arguments));
160  *     return p;
161  * };
162  *
163  * //This example creates an object with a world property but its not a function!
164  * var handle = comb.createFunctionWrapper({world : "world"}, createNewMyObject, createNewMyObject);
165  *
166  * handle.world //=> "world"
167  * var a = handle(1);
168  * a.hello;  //=>"hello"
169  * a.args; //=> [1];
170  * a = new handle(1,2);
171  * a.hello; //=>"hello"
172  * a.args; //=> [1,2];
173  * @param obj the object to proxy
174  * @param {Function} handler the handler to call when the object is used as a function
175  * @param {Function} constructTrap the funciton to use when using new on the object
176  * @param {Object} opts the prototype of the object.
177  * @memberOf comb
178  */
179 comb.createFunctionWrapper = function (obj, handler, constructTrap, opts) {
180     var args = misc.argsToArray(arguments), ret;
181     if (args.length != 4) {
182         opts = object.isHash(args[args.length - 1]) ? args.pop() : null;
183         constructTrap = functions.isFunction(args[args.length - 1]) ? args.pop() : null;
184         handler = functions.isFunction(args[args.length - 1]) ? args.pop() : null;
185     }
186     if (misc.isUndefined(obj)) throw new Error("obj required when using create function wrapper");
187     if (functions.isFunction(constructTrap) && !functions.isFunction(handler)) {
188         ret = Proxy.createFunction(handlerMaker(obj), constructTrap);
189     } else {
190         ret = Proxy.createFunction(handlerMaker(obj), handler, constructTrap);
191     }
192 
193 
194     if (opts) {
195         Proxy.setPrototype(ret, object.isHash(opts) ? opts : opts.prototype);
196     }
197     return ret;
198 };
