104         /**
105          * Traverse a tree until the callback function returns false
106          *
107          * <p><b>Not typically used directly</b></p>
108          *
109          * @param {Object} node the node to start at
110          * @param {Tree.PRE_ORDER|Tree.POST_ORDER|Tree.IN_ORDER|Tree.REVERSE_ORDER} [order=Tree.IN_ORDER] the traversal scheme
111          * @param {Function} callback called for each item, traversal continues until the function returns false
112          *
113          */
114         traverseWithCondition : function(node, order, callback) {
115             var cont = true;
116             if (node) {
117                 order = order || Tree.PRE_ORDER;
118                 if (order === Tree.PRE_ORDER) {
119                     cont = callback(node.data);
120                     if (cont) {
121                         cont = this.traverseWithCondition(node.left, order, callback);
122                         cont && (cont = this.traverseWithCondition(node.right, order, callback));
123 
124                     }
125                 } else if (order === Tree.IN_ORDER) {
126                     cont = this.traverseWithCondition(node.left, order, callback);
127                     if (cont) {
128                         cont = callback(node.data);
129                         cont && (cont = this.traverseWithCondition(node.right, order, callback));
130                     }
131                 } else if (order === Tree.POST_ORDER) {
132                     cont = this.traverseWithCondition(node.left, order, callback);
133                     if (cont) {
134                         cont && (cont = this.traverseWithCondition(node.right, order, callback));
135                         cont && (cont = callback(node.data));
136                     }
137                 } else if (order === Tree.REVERSE_ORDER) {
138                     cont = this.traverseWithCondition(node.right, order, callback);
139                     if (cont) {
140                         cont = callback(node.data);
141                         cont && (cont = this.traverseWithCondition(node.left, order, callback));
142                     }
143                 }
144             }
145             return cont;
146         },

148         /**
149          * Traverse a tree
150          *
151          * <p><b>Not typically used directly</b></p>
152          *
153          * @param {Object} node the node to start at
154          * @param {Tree.PRE_ORDER|Tree.POST_ORDER|Tree.IN_ORDER|Tree.REVERSE_ORDER} [order=Tree.IN_ORDER] the traversal scheme
155          * @param {Function} callback called for each item
156          *
157          */
158         traverse : function(node, order, callback) {
159             if (node) {
160                 order = order || Tree.PRE_ORDER;
161                 if (order === Tree.PRE_ORDER) {
162                     callback(node.data);
163                     this.traverse(node.left, order, callback);
164                     this.traverse(node.right, order, callback);
165                 } else if (order === Tree.IN_ORDER) {
166                     this.traverse(node.left, order, callback);
167                     callback(node.data);
168                     this.traverse(node.right, order, callback);
169                 } else if (order === Tree.POST_ORDER) {
170                     this.traverse(node.left, order, callback);
171                     this.traverse(node.right, order, callback);
172                     callback(node.data);
173                 } else if (order === Tree.REVERSE_ORDER) {
174                     this.traverseWithCondition(node.right, order, callback);
175                     callback(node.data);
176                     this.traverseWithCondition(node.left, order, callback);
177 
178                 }
179             }

354                 var cmp = this.compare(value, root.data);
355                 if (cmp) {

358                     ret = true;
