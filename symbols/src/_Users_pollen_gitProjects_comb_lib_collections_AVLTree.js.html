 16 var makeNode = function(data) {

 18         data : data,

 25 var rotateSingle = function(root, dir, otherDir) {
 26     var save = root[otherDir];
 27     root[otherDir] = save[dir];
 28     save[dir] = root;
 29     return save;
 30 };

 33 var rotateDouble = function(root, dir, otherDir) {
 34     root[otherDir] = rotateSingle(root[otherDir], otherDir, dir);
 35     return rotateSingle(root, dir, otherDir);
 36 };

 38 var adjustBalance = function(root, dir, bal) {
 39     var otherDir = dir == "left" ? "right" : "left";
 40     var n = root[dir], nn = n[otherDir];
 41     if (nn.balance == 0)
 42         root.balance = n.balance = 0;
 43     else if (nn.balance == bal) {
 44         root.balance = -bal;
 45         n.balance = 0;
 46     }
 47     else { /* nn.balance == -bal */
 48         root.balance = 0;
 49         n.balance = bal;
 50     }
 51     nn.balance = 0;
 52 };

 54 var insertAdjustBalance = function(root, dir) {
 55     var otherDir = dir == "left" ? "right" : "left";
 56 
 57     var n = root[dir];
 58     var bal = dir == "left" ? -1 : +1;
 59 
 60     if (n.balance == bal) {
 61         root.balance = n.balance = 0;
 62         root = rotateSingle(root, otherDir, dir);
 63     }
 64     else {
 65         adjustBalance(root, dir, bal);
 66         root = rotateDouble(root, otherDir, dir);
 67     }
 68 
 69     return root;
 70 
 71 };

 73 var removeAdjustBalance = function(root, dir, done) {
 74     var otherDir = dir == "left" ? "right" : "left";
 75     var n = root[otherDir];
 76     var bal = dir == "left" ? -1 : 1;
 77     if (!n) console.log(root, dir);
 78     if (n.balance == -bal) {
 79         root.balance = n.balance = 0;
 80         root = rotateSingle(root, dir, otherDir);
 81     }
 82     else if (n.balance == bal) {
 83         adjustBalance(root, otherDir, -bal);
 84         root = rotateDouble(root, dir, otherDir);
 85     }
 86     else { /* n.balance == 0 */
 87         root.balance = -bal;
 88         n.balance = bal;
 89         root = rotateSingle(root, dir, otherDir);
 90         done.done = true;
 91     }
 92     return root;

 95 var insert = function(root, data, done, compare) {
 96     if (root == null || root == undefined)
 97         root = makeNode(data);
 98     else {
 99         var dir = compare(data, root.data) == -1 ? "left" : "right";
100         root[dir] = insert(root[dir], data, done, compare);
101 
102         if (!done.done) {
103             /* Update balance factors */
104             root.balance += dir == "left" ? -1 : 1;
105             /* Rebalance as necessary and terminate */
106             if (root.balance == 0)
107                 done.done = true;
108             else if (abs(root.balance) > 1) {
109                 root = insertAdjustBalance(root, dir);
110                 done.done = true;
111             }
112         }
113     }
114 
115     return root;
116 };

118 var remove = function(root, data, done, compare) {
119     var dir, cmp, save, b;
120     if (root) {
121         //Remove node
122         cmp = compare(data, root.data);
123         if (cmp === 0) {
124             // Unlink and fix parent
125             var l = root.left, r = root.right;
126             if (!l || !r) {
127                 dir = !l ? "right" : "left";
128                 save = root[dir];
129                 return save;
130             }
131             else {
132                 var heir = l, r;
133                 while ((r = heir.right) != null) {
134                     heir = r;
135                 }
136                 root.data = heir.data;
137                 //reset and start searching
138                 data = heir.data;
139             }
140         }
141         dir = compare(root.data, data) == -1 ? "right" : "left";
142         root[dir] = remove(root[dir], data, done, compare);
143         if (!done.done) {
144             /* Update balance factors */
145             b = (root.balance += (dir == "left" ? 1 : -1));
146             /* Terminate or rebalance as necessary */
147             var a = abs(b);
148             if (a === 1)
149                 done.done = true;
150             else if (a > 1)
151                 root = removeAdjustBalance(root, dir, done);
152         }
153     }
154     return root;
155 };

158 /**
159  * @class <p>An AVL tree is a self-balancing binary search tree.
160  *    In an AVL tree, the heights of the two child subtrees of any node differ by at most one.
161  *    Lookup, insertion, and deletion all take O(log n) time in both the average and worst cases,
162  *    where n is the number of nodes in the tree prior to the operation.
163  *    Insertions and deletions may require the tree to be rebalanced by one or more tree rotations.</p>
164  * <p>AVL trees are more rigidly balanced than red-black trees, leading to slower insertion and removal but faster retrieval</p>
165  *
166  * <b>Performance</b>
167  * <table>
168  *     <tr><td></td><td>Best</td><td>Worst</td></tr>
169  *     <tr><td>Space</td><td>O(n)</td><td>O(n)</td></tr>
170  *     <tr><td>Search</td><td>O(log n)</td><td>O(log n)</td></tr>
171  *     <tr><td>Insert</td><td>O(log n)</td><td>O(log n)</td></tr>
172  *     <tr><td>Delete</td><td>O(log n)</td><td>O(log n)</td></tr>
173  * <table>
174  * @name AVLTree
175  * @augments comb.collections.Tree

177  */
