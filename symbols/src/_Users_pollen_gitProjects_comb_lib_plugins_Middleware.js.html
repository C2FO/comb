 14          * Object containing the hook methods

 16         __hooks : {pre : {}, post : {}},
 17 
 18         /**
 19          * Plugin to enable middleware on a class
 20          *
 21          * @example
 22          *
 23          * var Mammal = define(comb.plugins.Middleware, {
 24          *  instance : {
 25          *
 26          *    constructor: function(options) {
 27          *        options = options || {};
 28          *        this.super(arguments);
 29          *        this._type = options.type || "mammal";
 30          *    },
 31          *
 32          *    speak : function() {
 33          *        var ret = new comb.Promise();
 34          *        this._hook("pre", "speak")
 35          *                .then(comb.hitch(this, "_hook", "post", "speak"), hitch(ret, "errback"))
 36          *                .then(comb.hitch(ret, "callback"), comb.hitch(ret, "errback"));
 37          *        return ret;
 38          *    }
 39          *  }
 40          *});
 41          *
 42          *  Mammal.pre('speak', function(next){
 43          *     //do something meaningful
 44          *     next();
 45          *  });
 46          *  var m = new Mammal({color : "gold"});
 47          *  m.speak();
 48          *
 49          * @constructs
 50          */
 51         constructor : function() {
 52             this.__hooks = obj.merge({}, this.__hooks);
 53             this.super(arguments);
 54         },
 55 
 56         /**
 57          * <p>Protected!</p>
 58          *
 59          * <p>Call to initiate middleware for the topic</p>
 60          * <p><b>NOTE:</b> this function takes a variable number of arguments
 61          *       whatever comes after the op param will be passed into
 62          *       the listening function, with the last argument to the listenting
 63          *       function being the next function</p>
 64          *
 65          *
 66          * @public
 67          * @param {"pre"|"post"} state the state in which the hook should be called
 68          * @param {String} op the operation that is being acted upong
 69          * @param args arguments to be passed into the listening functions.
 70          * @returns {comb.Promise} a promise to use after middleware chain completes
 71          *
 72          */
 73         _hook : function(state, op, args) {
 74             args = args || [];
 75             var promise = new Promise();
 76             if (this.__hooks[state] && this.__hooks[state][op]) {
 77                 var funcs = this.__hooks[state][op], length = funcs.length;
 78                 if (length) {
 79                     var count = 0;
 80                     var next = func.hitch(this, function() {
 81                         //if Ive looped through all of them callback
 82                         if (count == length) {
 83                             promise.callback();
 84                         } else {
 85                             //call next
 86                             var nextArgs = args.slice(0);
 87                             nextArgs.unshift(next);
 88                             funcs[count++].apply(this, nextArgs);
 89                         }
 90                     });
 91                     next();
 92                 } else {
 93                     promise.callback();
 94                 }
 95             } else {
 96                 promise.callback();
 97             }
 98             return promise;
 99         },
100 
101         /**
102          * Use to listen to before an event occurred i.e. pre save
103          *
104          * <b>NOTE:</b></br>
105          * <ul>
106          *     <li>You must call next in order for the middleware chain to complete</li>
107          *      <li>This connects to events on the instance of an object, not all instances!</li>
108          *      <li>Hooks are called in the order they are received!</li>
109          *      <li> When connecting your callback will be called in the scope of the class</br>if you want a certain scope use {@link comb.hitch}</li>
110          *  </ul>
111          *
112          * @example
113          *      instance.pre("save", function(args,...., next){
114          *          //do something...
115          *          //you have to call next!!!!!
116          *          next();
117          *      });
118          *
119          * */
120         pre : function(fun, callback) {
121             var hook = this.__hooks.pre[fun];
122             if (!hook) {
123                 hook = this.__hooks.pre[fun] = [];
124             }
125             hook.push(callback);
126         },

132          *     <li>You must call next in order for the middleware chain to complete</li>
133          *      <li>This connects to events on the instance of an object, NOT all instances!</li>
134          *      <li>Hooks are called in the order they are received!</li>
135          *      <li>When connecting your callback will be called in the scope of the class</br>if you want a certain scope use {@link comb.hitch}</li>
136          *  </ul>
137          * @example
138          *
139          * instance.post("save", function(next){
140          *                //do something...
141          *                 //you have to call next!!!!!
142          *                 next();
143          *          });
144          * */
145         post : function(fun, callback) {
146             var hook = this.__hooks.post[fun];
147             //if I havent initialized it create it;
148             if (hook == undefined) {
149                 hook = this.__hooks.post[fun] = [];
150             }
151             hook.push(callback);
152         }
