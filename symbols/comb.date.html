    difference : function(date1, date2, interval, utc) {
        date2 = date2 || new Date();
        interval = interval || "day";
        var yearDiff = date2.getFullYear() - date1.getFullYear();
        var delta = 1;         switch (interval) {
            case "quarter":
            case "quarters":
                var m1 = date1[utc ? "getUTCMonth" : "getMonth"]();
                var m2 = date2[utc ? "getUTCMonth" : "getMonth"]();
                
                var q1 = floor(m1 / 3) + 1;
                var q2 = floor(m2 / 3) + 1;
                
                q2 += (yearDiff * 4);
                delta = q2 - q1;
                break;
            case "weekday":
            case "weekdays":

                var weeks = parseInt(date.difference(date1, date2, "week", utc));

                if (mod == 0) {

                                break;

                                adj = 0;
                                break;
                            
                            case bDay == 0:
                                adj = -2;
                                break;
                            
                            case (dayMark + mod) > 5:
                                adj = -2;
                        }
                    } else if (days < 0) {
                        switch (true) {
                            
                            case aDay == 6:
                                adj = 0;
                                break;
                            
                            case aDay == 0:
                                adj = 1;
                                break;
                            
                            case bDay == 6:
                                adj = 2;
                                break;
                            
                            case bDay == 0:
                                adj = 1;
                                break;
                            
                            case (dayMark + mod) < 0:
                                adj = 2;
                        }
                    }
                    days += adj;
                    days -= (weeks * 2);
                }
                delta = days;
                break;
            case "year":
            case "years":
                delta = yearDiff;
                break;

            case "months":
