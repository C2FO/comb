function (key){
   var heap = this.__heap;
   for (var i = heap.length - 1; i >= 0; i--) {
       if (heap[i].key == key) {
           return true;
       }
   }
   return false;
           
}

   var ret = undefined, heap = this.__heap, l = heap.length;
   if (l) {
       ret = heap[0];
