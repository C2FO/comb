113 var mixinSupers = function (sup, arr, bases) {
114     var meta = sup.__meta;
115     !meta && (meta = (sup.__meta = {}));
116     var unique = sup.__meta.unique;
117     !unique && (meta.unique = "define" + ++classCounter);
118     //check it we already have this super mixed into our prototype chain
119     //if true then we have already looped their supers!
120     if (bases.indexOf(unique) == -1) {
121         //add their id to our bases
122         bases.push(unique);
123         var supers = sup.__meta.supers || [], i = supers.length - 1 || 0;
124         while (i >= 0) {
125             mixinSupers(supers[i--], arr, bases);
126         }

199             f:null,
200             pos:0,
201             name:null
202         }
203     };

206         supers = Array.isArray(sup) ? sup : [sup];
207         sup = supers.shift();
208         child.__proto__ = sup;
209         childProto.__proto__ = sup.prototype;
210         mixinSupers(sup.prototype, instanceSupers, bases),
211             mixinSupers(sup, staticSupers, staticBases);
212     }
213     if (proto) {
214         var instance = meta.proto = proto.instance || {};
215         !instance.hasOwnProperty("constructor") && (instance.constructor = defaultFunction);
216         var stat = childMeta.proto = proto.static || {};
217         stat.init = stat.init || defaultFunction;

219         defineProps(child, stat, true);
220     } else {
221         meta.proto = {};

224         childProto.constructor = functionWrapper(defaultFunction, "constructor");

236 /**

238  *
239  * <p>
240  *     Class methods
241  *     <ul>
242  *         <li>as(module | bject, name): exports the object to module or the object with the name</li>
243  *         <li>mixin(mixin) : mixes in an object</li>
244  *     </ul>
245  *     </br>
246  *     Instance methods
247  *     <ul>
248  *         <li>_super(argumnents, [?newargs]): calls the super of the current method</li>
249  *     </ul>
250  *
251  *      </br>
252  *     Instance properties
253  *     <ul>
254  *         <li>_static: use to reference class properties and methods</li>
255  *     </ul>
256  *
257  * </p>
258  *
259  *
260  * @example
261  *  //Class without a super class
262  * var Mammal = comb.define(null, {
263  *      instance : {
264  *
265  *          constructor: function(options) {
266  *              options = options || {};
267  *              this._super(arguments);

277  *              type : function() {
278  *                  return this._type;
279  *              }

281  *
282  *           //Define your setters
283  *          setters : {
284  *              type : function(t) {
285  *                  this._type = t;
286  *              }
287  *          }
288  *      },

466     };

471 /**
472  * Defines a singleton instance of a Class
473  * @example
474  *  var MyLab = comb.singleton([Mammal, Wolf, Dog, Breed]);
475  *  var myLab1 = new MyLab();
476  *  myLab1.type = "collie"
477  *  var myLab2 = new MyLab();
478  *  myLab1 === myLab2 => true
479  *  myLab1.type => "collie"
480  *  myLab2.type => "collie"
481  * See {@link define}
482  */
483 exports.singleton = function (sup, proto) {
484     var retInstance;
485     var child = function () {
486         if (!retInstance) {
487             this.constructor.apply(this, arguments);
488             retInstance = this;
489         }
490         return retInstance;
491     };
492     __define(child, sup, proto);
493     return  child.init() || child;
494 };
