1 var func = require("../base/functions"),
  2         define = require("../define").define;

  5 var Broadcaster = define(null, {
  6     instance : {
  7     /** @lends comb.plugins.Broadcaster.prototype */
  8         /**
  9          * Plugin to allow a class to easily broadcast events
 10          *
 11          * @example
 12          *
 13          * var Mammal = define(comb.plugins.Broadcaster, {
 14          *   instance : {
 15          *
 16          *      constructor: function(options) {
 17          *          options = options || {};
 18          *          this.super(arguments);
 19          *          this._type = options.type || "mammal";
 20          *      },
 21          *
 22          *      speak : function() {
 23          *          var str = "A mammal of type " + this._type + " sounds like";
 24          *          this.broadcast("speak", str);
 25          *          this.onSpeak(str);
 26          *          return str;
 27          *      },
 28          *
 29          *      onSpeak : function(){}
 30          *    }
 31          * });
 32          *
 33          *
 34          * var m = new Mammal({color : "gold"});
 35          * m.listen("speak", function(str){
 36          *      //called back from the broadcast event
 37          *       console.log(str);
 38          * });
 39          * m.speak();
 40          *
 41          * @constructs
 42          */
 43         constructor : function() {
 44             this.__listeners = {};
 45         },
 46 
 47         /**
 48          * Broadcasts an event from an object
 49          *
 50          * @param name the name of the event to broadcast
 51          * @param {Object|String|Function|Date|Number} [args]  variable number of arguments to pass to listeners, can be anything
 52          */
 53         broadcast : function(name, args) {
 54             var args = Array.prototype.slice.call(arguments);
 55             var topic = args.splice(0, 1)[0];
 56             if (topic) {
 57                 var list = this.__listeners[topic];
 58                 if (list) {
 59                     for (var i = list.length - 1; i >= 0; i--) {
 60                         var han = list[i], cb = han.cb;
 61                         if (cb) {
 62                             cb.apply(this, args);
 63                         }
 64                     }
 65                 }
 66             }
 67         },
 68 
 69         /**
 70          * Listens to a broadcasted event
 71          * Simimlar to {@link comb.listen}
 72          *
 73          * @param {String} topic the topic to listen to
 74          * @param {Function} callback the function to callback on event publish
 75          *
 76          * @returns {Array} handle to disconnect a topic
 77          */
 78         listen : function(topic, callback) {
 79             if (!func.isFunction(callback)) throw new Error("callback must be a function");
 80             var handle = {
 81                 topic : topic,
 82                 cb : callback,
 83                 pos : null
 84             };
 85             var list = this.__listeners[topic];
 86             if (!list) {
 87                 list = (this.__listeners[topic] = []);
 88             }
 89             list.push(handle);
 90             handle.pos = list.length - 1;
 91             return handle;
 92         },
 93 
 94         /**
 95          * Disconnects a listener
 96          * Similar to {@link comb.unListen}
 97          *
 98          * @param  handle disconnect a handle returned from Broadcaster.listen
 99          */
100         unListen : function(handle) {
101             if (handle) {
102                 var listeners = this.__listeners;
103                 var topic = handle.topic, list = listeners[topic];
104                 if (list) {
105                     for (var i = list.length - 1; i >= 0; i--) {
106                         if (list[i] == handle) {
107                             list.splice(i, 1);
108                         }
109                     }
110                     if (!list.length) {
111                         delete listeners[topic];
112                     }
113                 }
114             }
115         }
116     }
117 });
