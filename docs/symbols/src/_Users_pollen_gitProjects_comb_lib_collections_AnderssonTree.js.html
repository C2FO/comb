 16 var makeNode = function(data, level) {
 17     return {
 18         data : data,
 19         level : level,
 20         left : nil,
 21         right : nil
 22     }

 25 var skew = function(root) {
 26     if (root.level != 0 && root.left.level == root.level) {
 27         var save = root.left;
 28         root.left = save.right;
 29         save.right = root;
 30         root = save;
 31     }
 32     return root;
 33 };
 34 
 35 var split = function(root) {
 36     if (root.level != 0 && root.right.right.level == root.level) {
 37         var save = root.right;
 38         root.right = save.left;
 39         save.left = root;
 40         root = save;
 41         ++root.level;
 42     }
 43     return root;
 44 };

 46 var insert = function(root, data, compare) {
 47     if (root == nil) {
 48         root = makeNode(data, 1);
 49     }
 50     else {
 51         var dir = compare(data, root.data) == -1 ? "left" : "right";
 52         root[dir] = insert(root[dir], data, compare);
 53         root = skew(root);
 54         root = split(root);
 55     }
 56     return root;
 57 };
 58 
 59 var remove = function(root, data, compare) {
 60     var rLeft, rRight;
 61     if (root != nil) {
 62         var cmp = compare(data, root.data);
 63         if (cmp == 0) {
 64             rLeft = root.left,rRight = root.right;
 65             if (rLeft != nil && rRight != nil) {
 66                 var heir = rLeft;
 67                 while (heir.right != nil)
 68                     heir = heir.right;
 69                 root.data = heir.data;
 70                 root.left = remove(rLeft, heir.data, compare);
 71             } else {
 72                 root = root[rLeft == nil ? "right" : "left"];
 73             }
 74         } else {
 75             var dir = cmp == -1 ? "left" : "right";
 76             root[dir] = remove(root[dir], data, compare);
 77         }
 78     }
 79     if (root != nil) {
 80         var rLevel = root.level;
 81         var rLeftLevel = root.left.level, rRightLevel = root.right.level;
 82         if (rLeftLevel < rLevel - 1 || rRightLevel < rLevel - 1) {
 83             if (rRightLevel > --root.level)
 84                 root.right.level = root.level;
 85 
 86             root = skew(root);
 87             root = split(root);
 88         }
 89     }
 90     return root;
 91 };
