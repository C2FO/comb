 21 var parseProperties = function (properties) {
 22     for (var i in properties) {
 23         var logger = Logger.getLogger(i);
 24         var props = properties[i], level = props.level, appenderArr = props.appenders;
 25         if (level) {
 26             level = Level.toLevel(level);
 27             if (level) {
 28                 logger.level = level;
 29             }
 30         }
 31         if (appenderArr && base.isArray(appenderArr)) {
 32             for (var j = appenderArr.length - 1; j >= 0; j--) {
 33                 var appenderProps = appenderArr[j], type = appenderProps.type;
 34                 appenderProps.type = null;
 35                 if (type) {
 36                     var appender;
 37                     switch (type.toLowerCase()) {
 38                         case "consoleappender":
 39                             appender = appenders.ConsoleAppender;
 40                             break;
 41                         case "jsonappender":
 42                             appender = appenders.JSONAppender;
 43                             break;
 44                         case "fileappender":
 45                             appender = appenders.FileAppender;
 46                             break;
 47                         case "rollingfileappender":
 48                             appender = appenders.RollingFileAppender;
 49                             break;
 50                     }
 51                     if (appender) {
 52                         logger.addAppender(new appender(appenderProps));
 53                     }
 54 
 55                 }
 56             }
 57         }
 58     }
 59     checkProcessUncaughtException();
 60 };

 63 /**
 64  * @class default configurator for logging
 65  *
 66  * @name BasicConfigurator
 67  * @memberOf comb.logging
 68  *
 69  */
 70 var BasicConfigurator = (exports.BasicConfigurator = define(null, {
 71     instance:{
 72         /**@lends comb.logging.BasicConfigurator.prototype*/
 73 
 74         constructor:function () {
 75             if (!Logger) {
 76                 logging = require("./index").logging;
 77                 Logger = logging.Logger;
 78                 Level = logging.Level;
 79                 appenders = logging.appenders;
 80             }
 81         },
 82 
 83         /**
 84          * Configure logging.
 85          *
 86          * @param {comb.logging.Appender} [appender=null] appender to add to the root logger, by default a console logger is added.
 87          */
 88         configure:function (appender) {
 89             var rootLogger = Logger.getRootLogger();
 90             rootLogger.removeAllAppenders();
 91             if (base.isInstanceOf(appender, appenders.Appender)) {
 92                 rootLogger.addAppender(appender);
 93             } else {
 94                 rootLogger.addAppender(new appenders.ConsoleAppender());
 95             }
 96             checkProcessUncaughtException();
 97         }
 98     }
 99 }));

101 /**
102  * @class Configures comb.Logger with the properties or properties contained within a file
103  *
104  * @example
105  *
106  * var propertyConfigurator = new comb.logging.PropertyConfigurator();
107  *
108  * propertyConfigurator.configure("/location/of/combLogger.json");
109  *
110  * //or
111  *
112  * var config = {
113  *     "my.logger" : {
114  *         level : "INFO",
115  *         appenders : [
116  *             {
117  *                 //default file appender
118  *                 type : "FileAppender",
119  *                 file : "/var/log/myApp.log",
120  *             },
121  *             {
122  *                 //default JSON appender
123  *                 type : "JSONAppender",
124  *                 file : "/var/log/myApp.JSON",
125  *             },
126  *             {
127  *                 type : "FileAppender",
128  *                  //override default patter
129  *                 pattern : "{[EEEE, MMMM dd, yyyy h:m a]timeStamp} {[5]level}"
130  *                          + " {[- 5]levelName} {[-20]name} : {message}",
131  *                 //location of my log file
132  *                 file : "/var/log/myApp-errors.log",
133  *                 //override name so it will get added to the log
134  *                 name : "errorFileAppender",
135  *                 //overwrite each time
136  *                 overwrite : true,
137  *                 //explicity set the appender to only accept errors
138  *                 level : "ERROR"
139  *             },
140  *             {
141  *                 type : "JSONAppender",
142  *                 file : "/var/log/myApp-error.json",
143  *                 //explicity set the appender to only accept errors
144  *                 level : "ERROR"
145  *             }
146  *         ]
147  *     }
148  *     //repeat for more loggers
149  *
150  *     propertyConfigurator.configure(config);
151  * }
152  *
153  * @name PropertyConfigurator
154  * @augments comb.logging.BasicConfigurator
155  * @memberOf comb.logging
156  *
157  */
158 exports.PropertyConfigurator = define(BasicConfigurator, {
159     instance:{
160         /**@lends comb.logging.PropertyConfigurator.prototype*/
161         /**
162          * Call to configure logging
163          *
164          * @example
165          *
166          * //Example configuration
167          *  {
168          *     "my.logger" : {
169          *         level : "INFO",
170          *         appenders : [
171          *             {
172          *                 //default file appender
173          *                 type : "FileAppender",
174          *                 file : "/var/log/myApp.log",
175          *             },
176          *             {
177          *                 //default JSON appender
178          *                 type : "JSONAppender",
179          *                 file : "/var/log/myApp.JSON",
180          *             },
181          *             {
182          *                 type : "FileAppender",
183          *                  //override default patter
184          *                 pattern : "{[EEEE, MMMM dd, yyyy h:m a]timeStamp} {[5]level}"
185          *                          + " {[- 5]levelName} {[-20]name} : {message}",
186          *                 //location of my log file
187          *                 file : "/var/log/myApp-errors.log",
188          *                 //override name so it will get added to the log
189          *                 name : "errorFileAppender",
190          *                 //overwrite each time
191          *                 overwrite : true,
192          *                 //explicity set the appender to only accept errors
193          *                 level : "ERROR"
194          *             },
195          *             {
196          *                 type : "JSONAppender",
197          *                 file : "/var/log/myApp-error.json",
198          *                 //explicity set the appender to only accept errors
199          *                 level : "ERROR"
200          *             }
201          *         ]
202          *     }
203          *
204          * @param {Object|String} properties Object containing configuration or string containing a file name with the configuration.
205          */
206         configure:function (properties) {
207             var rootLogger = Logger.getRootLogger();
208             rootLogger.removeAllAppenders();
209             if (base.isHash(properties)) {
210                 parseProperties(properties);
211             } else {
212                 fs.readFile(properties, function (err, res) {
213                     if (err) {
214                         throw err;
215                     } else {
216                         try {
217                             parseProperties(JSON.parse(res));
218                         } catch (e) {
219                             throw e;
220                         }
221                     }
222                 });
223             }
224         }
225     }
226 });
