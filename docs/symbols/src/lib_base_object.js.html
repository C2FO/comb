  9 var _deepEqual = function (actual, expected) {
 10     // 7.1. All identical values are equivalent, as determined by ===.
 11     if (actual === expected) {
 12         return true;
 13 
 14     } else if (Buffer.isBuffer(actual) && Buffer.isBuffer(expected)) {
 15         if (actual.length != expected.length) return false;
 16 
 17         for (var i = 0; i < actual.length; i++) {
 18             if (actual[i] !== expected[i]) return false;
 19         }
 20 
 21         return true;
 22 
 23         // 7.2. If the expected value is a Date object, the actual value is
 24         // equivalent if it is also a Date object that refers to the same time.
 25     } else if (actual instanceof Date && expected instanceof Date) {
 26         return actual.getTime() === expected.getTime();
 27 
 28         // 7.3 If the expected value is a RegExp object, the actual value is
 29         // equivalent if it is also a RegExp object with the same source and
 30         // properties (`global`, `multiline`, `lastIndex`, `ignoreCase`).
 31     } else if (actual instanceof RegExp && expected instanceof RegExp) {
 32         return actual.source === expected.source &&
 33             actual.global === expected.global &&
 34             actual.multiline === expected.multiline &&
 35             actual.lastIndex === expected.lastIndex &&
 36             actual.ignoreCase === expected.ignoreCase;
 37 
 38         // 7.4. Other pairs that do not both pass typeof value == 'object',
 39         // equivalence is determined by ==.
 40     } else if (typeof actual != 'object' && typeof expected != 'object') {
 41         return actual == expected;
 42 
 43         // 7.5 For all other Object pairs, including Array objects, equivalence is
 44         // determined by having the same number of owned properties (as verified
 45         // with Object.prototype.hasOwnProperty.call), the same set of keys
 46         // (although not necessarily the same order), equivalent values for every
 47         // corresponding key, and an identical 'prototype' property. Note: this
 48         // accounts for both named and indexed properties on Arrays.
 49     } else {
 50         return objEquiv(actual, expected);
 51     }
 52 };

 55 var objEquiv = function (a, b) {
 56     if (isUndefinedOrNull(a) || isUndefinedOrNull(b))
 57         return false;
 58     // an identical 'prototype' property.
 59     if (a.prototype !== b.prototype) return false;
 60     //~~~I've managed to break Object.keys through screwy arguments passing.
 61     //   Converting to array solves the problem.
 62     if (isArguments(a)) {
 63         if (!isArguments(b)) {
 64             return false;
 65         }
 66         a = pSlice.call(a);
 67         b = pSlice.call(b);
 68         return _deepEqual(a, b);
 69     }
 70     try {
 71         var ka = Object.keys(a),
 72             kb = Object.keys(b),
 73             key, i;
 74     } catch (e) {//happens when one is a string literal and the other isn't
 75         return false;
 76     }
 77     // having the same number of owned properties (keys incorporates
 78     // hasOwnProperty)
 79     if (ka.length != kb.length)
 80         return false;
 81     //the same set of keys (although not necessarily the same order),
 82     ka.sort();
 83     kb.sort();
 84     //~~~cheap key test
 85     for (i = ka.length - 1; i >= 0; i--) {
 86         if (ka[i] != kb[i])
 87             return false;
 88     }
 89     //equivalent values for every corresponding key, and
 90     //~~~possibly expensive deep test
 91     for (i = ka.length - 1; i >= 0; i--) {
 92         key = ka[i];
 93         if (!_deepEqual(a[key], b[key])) return false;
 94     }
 95     return true;
 96 }

 98 var merge = function (target, source) {
 99     var name, s;
100     for (name in source) {
101         s = source[name];
102         if (!(name in target) || (target[name] !== s)) {
103             target[name] = s;
104         }
105     }
106     return target;
107 };

109 var deepMerge = function (target, source) {
110     var name, s, t;
111     for (name in source) {
112         s = source[name], t = target[name];
113         if (!_deepEqual(t, s)) {
114             if (comb.isHash(t) && comb.isHash(s)) {
115                 target[name] = deepMerge(t, s);
116             } else if (comb.isHash(s)) {
117                 target[name] = deepMerge({}, s);
118             } else {
119                 target[name] = s;
120             }
121         }
122     }
123     return target;
124 };

126 /**
127  * Merges objects together
128  * NOTE: this function takes a variable number of objects to merge
129  *
130  * @example
131  *
132  * var myObj = {};
133  * comb.merge(myObj, {test : true});
134  *
135  * myObj.test => true
136  *
137  * comb.merge(myObj, {test : false}, {test2 : false}, {test3 : "hello", test4 : "world"});
138  * myObj.test => false
139  * myObj.test2 => false
140  * myObj.test3 => "hello"
141  * myObj.test4 => "world"
142  *
143  *
144  * @param {Object} obj the object to merge into
145  * @param {Object} props variable number of objects to merge into the obj
146  *
147  * @returns {Object} the merged object
148  */
149 comb.merge = function (obj, props) {
150     if (!obj) {
151         obj = {};
152     }
153     for (var i = 1, l = arguments.length; i < l; i++) {
154         merge(obj, arguments[i]);
155     }
156     return obj; // Object
157 };

159 /**
160  * Merges objects together only overriding properties that are different.
161  * NOTE: this function takes a variable number of objects to merge
162  *
163  * @example
164  *
165  * var myObj = {my : {cool : {property1 : 1, property2 : 2}}};
166  * comb.deepMerge(myObj, {my : {cool : {property3 : 3}}});
167  *
168  * myObj.my.cool.property1 => 1
169  * myObj.my.cool.property2 => 2
170  * myObj.my.cool.property3 => 3
171  *
172  *
173  * @param {Object} obj the object to merge into
174  * @param {Object} props variable number of objects to merge into the obj
175  *
176  * @returns {Object} the merged object
177  */
178 comb.deepMerge = function (obj, props) {
179     if (!obj) {
180         obj = {};
181     }
182     for (var i = 1, l = arguments.length; i < l; i++) {
183         deepMerge(obj, arguments[i]);
184     }
185     return obj; // Object
186 };

188 /**
189  * Extends the prototype of an object if it exists otherwise it extends the object.
190  *
191  * @example
192  *
193  *  var MyObj = function(){};
194  *  MyObj.prototype.test = true;
195  *  comb.extend(MyObj, {test2 : false, test3 : "hello", test4 : "world"});
196  *
197  *  var myObj = new MyObj();
198  *
199  *  myObj.test => true
200  *  myObj.test2 => false
201  *  myObj.test3 => "hello"
202  *  myObj.test4 => "world"
203  *
204  *  var myObj2 = {};
205  *  myObj2.test = true;
206  *  comb.extend(myObj2, {test2 : false, test3 : "hello", test4 : "world"});
207  *
208  *  myObj2.test => true
209  *  myObj2.test2 => false
210  *  myObj2.test3 => "hello"
211  *  myObj2.test4 => "world"
212  *
213  *
214  * @param {Object} parent the parent object to extend
215  * @param {Object} extend the extension object to mixin to the parent
216  *
217  * @returns {Object} returns the extended object
218  */
219 comb.extend = function (parent, extend) {
220     var proto = parent.prototype || parent;
221     return exports.merge(proto, extend);
222 };

236 /**
237  * Determins if an object is just a hash and not a qualified Object such as number
238  *
239  * @example
240  *    comb.isHash({}) => true
241  *    comb.isHash({1 : 2, a : "b"}) => true
242  *    comb.isHash(new Date()) => false
243  *    comb.isHash(new String()) => false
244  *    comb.isHash(new Number()) => false
245  *    comb.isHash(new Boolean()) => false
246  *    comb.isHash() => false
247  *    comb.isHash("") => false
248  *    comb.isHash(1) => false
249  *    comb.isHash(false) => false
250  *    comb.isHash(true) => false
251  * @param {Anything} obj the thing to test if it is a hash
252  *
253  * @returns {Boolean} true if it is a hash false otherwise
254  */
255 comb.isHash = function (obj) {
256     var ret = comb.isObject(obj);
257     return ret && obj.constructor === Object;
258 }

260 /**
261  * Determines if an object is empty
262  *
263  * @example
264  *
265  * comb.isEmpty({}) => true
266  * comb.isEmpty({a : 1}) => false
267  *
268  * @param object the object to test
269  * @returns {Boolean} true if the object is empty;
270  */
271 comb.isEmpty = function (object) {
272     if (comb.isObject(object)) {
273         for (var i in object) {
274             if (object.hasOwnProperty(i)) {
275                 return false;
276             }
277         }
278     }
279     return true;
280 };
