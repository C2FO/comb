  9 comb.string = {
 10     /**
 11      * Pads a string
 12      *
 13      * @example
 14      *
 15      * comb.string.pad("STR", 5, " ", true) => "STR  "
 16      * comb.string.pad("STR", 5, "$") => "$$STR"
 17      *
 18      * @param {String} string the string to pad
 19      * @param {Number} length the length of the string when padded
 20      * @param {String} ch character to pad the string with
 21      * @param {Boolean} [end=false] if true then the padding is added to the end
 22      *
 23      * @returns {String} the padded string
 24      */
 25     pad : function(string, length, ch, end) {
 26         string = "" + string; //check for numbers
 27         var strLen = string.length;
 28         while (strLen < length) {
 29             if (end) {
 30                 string += ch;
 31             } else {
 32                 string = ch + string;
 33             }
 34             strLen++;
 35         }
 36         return string;
 37     },
 38 
 39     /**
 40      * Formats a string with the specified format
 41      *
 42      * @example
 43      *
 44      *  comb.string.format("%s, %s", ["Hello", "World"]);
 45      *  comb.string.format("%s, %s", "Hello", "World");
 46      *  comb.string.format("{hello}, {world}", {hello : "Hello", world : {"World"});
 47      *
 48      * @param {String} str the string to format
 49      * @param {Object|Array|Arguments...} obj the parameters to replace in the string
 50      *                                    if an array is passed then the array is used sequentially
 51      *                                    if an object is passed then the object keys are used
 52      *                                    if a variable number of args are passed then they are used like an array
 53      *
 54      * @returns {String} the formatted string
 55      */
 56     format : function(str, obj) {
 57         if (obj instanceof Array) {
 58             var i = 0, len = obj.length;
 59             return str.replace(/%s/g, function() {
 60                 if (i < len) {
 61                     return obj[i++] || "";
 62                 } else {
 63                     return "%s";
 64                 }
 65             });
 66         } else if (typeof obj == "object") {
 67             return str.replace(/{(\w+)}/g, function(m, g) {
 68                 return obj[g] || "";
 69             });
 70         } else {
 71             var args = Array.prototype.slice.call(arguments).slice(1);
 72             return exports.string.format(str, args);
 73         }
 74     },
 75 
 76     /**
 77      * Converts a string to an array
 78      *
 79      * @example
 80      *
 81      * comb.string.toArray("a|b|c|d", "|") => ["a","b","c","d"]
 82      * comb.string.toArray("a", "|") => ["a"]
 83      * comb.string.toArray("", "|") => []
 84      *
 85      * @param {String} str the string to parse
 86      * @param {String} delimeter the delimeter to use
 87      */
 88     toArray : function(testStr, delim) {
 89         var ret = [];
 90         if (testStr) {
 91             if (testStr.indexOf(delim) > 0) return testStr.replace(/\s+/g, "").split(delim);
 92             else return [testStr];
 93         }
 94         return ret;
 95     }
