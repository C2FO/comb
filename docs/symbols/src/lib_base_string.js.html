  8 var FORMAT_REGEX = /%([^%|\s]*|(?:\[[^\[|\]]*\]))?([sjdDZ])/g;
  9 var INTERP_REGEX = /{(?:\[([^\[|\]]*)\])?(\w+)}/g;
 10 var STR_FORMAT = /(-?)(\+?)([A-Z|a-z|\W]?)([1-9][0-9]*)?$/;
 11 var OBJECT_FORMAT = /([1-9][0-9]*)$/g;

 13 var formatString = function(string, format) {
 14 	var match = format.match(STR_FORMAT), ret = string, cString = comb.string;
 15 	if (match) {
 16 		var isLeftJustified = match[1], padChar = match[3], width = match[4];
 17 		if (width) {
 18 			width = parseInt(width);
 19 			if (ret.length < width) {
 20 				ret = cString.pad(ret, width, padChar, isLeftJustified);
 21 			} else {
 22 				ret = cString.truncate(ret, width);
 23 			}
 24 		}
 25 	}
 26 	return ret;
 27 };
 28 
 29 var formatNumber = function(number, format) {
 30 	if (typeof number == "number") {
 31 		var cString = comb.string, ret = "" + number;
 32 		var match = format.match(STR_FORMAT);
 33 		if (match) {
 34 			var isLeftJustified = match[1], signed = match[2], padChar = match[3], width = match[4];
 35 			if (signed) {
 36 				ret = (number > 0 ? "+" : "") + ret;
 37 			}
 38 			if (width) {
 39 				width = parseInt(width);
 40 				if (ret.length < width) {
 41 					ret = cString.pad(ret, width, padChar || "0", isLeftJustified);
 42 				} else {
 43 					ret = cString.truncate(ret, width);
 44 				}
 45 			}
 46 
 47 		}
 48 	} else {
 49 		throw new Error("comb.string.format : when using %d the parameter must be a number!");
 50 	}
 51 	return ret;
 52 };

 54 var formatObject = function(object, format) {
 55 	var ret,  match = format.match(OBJECT_FORMAT), spacing = 0;
 56 	if (match) {
 57 		spacing = parseInt(match[0]);
 58 		if (isNaN(spacing)) spacing = 0;
 59 	}
 60 	try {
 61 		ret = JSON.stringify(object, null, spacing);
 62 	} catch(e) {
 63 		throw new Error("comb.string.format : Unable to parse json from ", object);
 64 	}
 65 	return ret;
 66 };

861 /**@namespace String utilities*/
862 comb.string = {
863 	/**
864 	 * Pads a string
865 	 *
866 	 * @example
867 	 *
868 	 * comb.string.pad("STR", 5, " ", true) => "STR  "
869 	 * comb.string.pad("STR", 5, "$") => "$$STR"
870 	 *
871 	 * @param {String} string the string to pad
872 	 * @param {Number} length the length of the string when padded
873 	 * @param {String} [ch= " "] character to pad the string with
874 	 * @param {Boolean} [end=false] if true then the padding is added to the end
875 	 *
876 	 * @returns {String} the padded string
877 	 */
878 	pad : function(string, length, ch, end) {
879 		string = "" + string; //check for numbers
880 		ch = ch || " ";
881 		var strLen = string.length;
882 		while (strLen < length) {
883 			if (end) {
884 				string += ch;
885 			} else {
886 				string = ch + string;
887 			}
888 			strLen++;
889 		}
890 		return string;
891 	},
892 
893 	/**
894 	 * Truncates a string to the specified length.
895 	 * @example
896 	 *
897 	 * //from the beginning
898 	 * comb.string.truncate("abcdefg", 3) => "abc";
899 	 * //from the end
900 	 * comb.string.truncate("abcdefg", 3,true) => "efg"
901 	 * //omit the length
902 	 * comb.string.truncate("abcdefg") => "abcdefg"
903 	 *
904 	 * @param {String} string the string to truncate
905 	 * @param {Number} [length = -1] the max length of the string, if the string is
906 	 *                              shorter than the length then the string is returned.
907 	 * @param {Boolean} [end=false] truncate starting at the end of the string
908 	 *

919 					ret = string.substring(0, length);

921 			}
922 		} else {

924 		}

928 	/**
929 	 * Formats a string with the specified format
930 	 *
931 	 * @example
932 	 *
933 	 *  var format = comb.string.format;
934 	 *
935 	 *  format("%s, %s", ["Hello", "World"]) => "Hello, World";
936 	 *  format("%[ 10]s, %[- 10]s", ["Hello", "World"])
937 	 *      => "     Hello, World     ";
938 	 *  format("%-!10s, %#10s, %10s and %-10s",
939 	 *                          "apple", "orange", "bananas", "watermelons")
940 	 *      => "apple!!!!!, ####orange,    bananas and watermelon"
941 	 *  format("%+d, %+d, %10d, %-10d, %-+#10d, %10d",
942 	 *                          1,-2, 1, 2, 3, 100000000000)
943 	 *      => "+1, -2, 0000000001, 2000000000, +3########, 1000000000"
944 	 *
945 	 *
946      *  format("%[h:mm a]D", [date]) => 7:32 PM - local -5
947      *  format("%[h:mm a]Z", [date]) => 12:32 PM - UTC
948      *
949      *  //When using object formats they must be in an array otherwise
950 	 *  //format will try to interpolate the properties into the string.
951 	 *  format("%j", [{a : "b"}])
952 	 *      => '{"a":"b"}'
953 	 *  format("%1j, %4j", [{a : "b"}, {a : "b"}])
954 	 *      => '{\n "a": "b"\n},\n{\n    "a": "b"\n}'
955 	 *  format("{hello}, {world}", {hello : "Hello", world : "World")
956 	 *      => "Hello, World";
957 	 *  format({[-s10]apple}, {[%#10]orange}, {[10]banana} and {[-10]watermelons}",
958 	 *                      {
959 	 *                          apple : "apple",
960 	 *                          orange : "orange",
961 	 *                          banana : "bananas",
962 	 *                          watermelons : "watermelons"
963 	 *        });
964 	 *      => applesssss, ####orange,    bananas and watermelon
965 	 *
966 	 * @param {String} str the string to format, if you want to use a spacing character as padding (other than \\s) then put your format in brackets.
967 	 *  <ol>
968 	 *      <li>String Formats %[options]s</li>
969 	 *          <ul>
970 	 *              <li>- : left justified</li>
971 	 *              <li>Char : padding character <b>Excludes d,j,s</b></li>
972 	 *              <li>Number : width</li>
973 	 *          </ul>
974 	 *      </li>
975 	 *      <li>Number Formats %[options]d</li>
976 	 *          <ul>
977 	 *              <li>- : left justified</li>
978 	 *              <li>+ : signed number</li>
979 	 *              <li>Char : padding character <b>Excludes d,j,s</b></li>
980 	 *              <li>Number : width</li>
981 	 *          </ul>
982 	 *      </li>
983 	 *      <li>Object Formats %[options]j</li>
984 	 *          <ul>
985 	 *              <li>Number : spacing for object properties.</li>
986 	 *          </ul>
987 	 *      </li>
988 	 *  </ol>
989 	 *
990 	 *
991 	 * @param {Object|Array|Arguments...} obj the parameters to replace in the string
992 	 *                                    if an array is passed then the array is used sequentially
993 	 *                                    if an object is passed then the object keys are used
994 	 *                                    if a variable number of args are passed then they are used like an array
995 	 *
996 	 * @returns {String} the formatted string
997 	 */
998 	format : function(str, obj) {
999 		!date && (date = require("./date"));
1000 		if (obj instanceof Array) {
1001 			var i = 0, len = obj.length;
1002 			//find the matches
1003 			return str.replace(FORMAT_REGEX, function(m, format, type) {
1004 				var replacer, ret;
1005 				if (i < len) {
1006 					replacer = obj[i++];
1007 				} else {
1008 					//we are out of things to replace with so
1009 					//just return the match?
1010 					return m;
1011 				}
1012 				if (m == "%s" || m == "%d" || m == "%D") {
1013 					//fast path!
1014 					ret = replacer;
1015 				} else if (m == "%j") {
1016 					try {
1017 						ret = JSON.stringify(replacer);
1018 					} catch(e) {
1019 						throw new Error("comb.string.format : Unable to parse json from ", replacer);
1020 					}
1021 				} else {
1022 					format = format.replace(/^\[|\]$/g, "");
1023 					switch (type) {
1024 						case "s":
1025 							ret = formatString(replacer, format);
1026 							break;
1027 						case "d":
1028 							ret = formatNumber(replacer, format);
1029 							break;
1030 						case "j":
1031 							ret = formatObject(replacer, format);
1032 							break;
1033 						case "D":
1034 							ret = date.date.format(replacer, format);
1035 							break;
1036                         case "Z":
1037 							ret = date.date.format(replacer, format, true);
1038 							break;
1039 					}
1040 				}
1041 				return ret;
1042 			});
1043 		} else if (typeof obj == "object") {
1044 			return str.replace(INTERP_REGEX, function(m, format, value) {
1045 				value = obj[value];
1046 				if (format) {
1047 					if (comb.isString(value)) {
1048 						return formatString(value, format);
1049 					} else if (typeof value == "number") {
1050 						return formatNumber(value, format);
1051 					} else if (date.isDate(value)) {
1052 						return date.date.format(value, format);
1053 					} else if (typeof value == "object") {
1054 						return formatObject(value, format);
1055 					}
1056 				} else {
1057 					return "" + value;
1058 				}
1059 			});
1060 		} else {
1061 			var args = Array.prototype.slice.call(arguments).slice(1);
1062 			return exports.string.format(str, args);
1063 		}
1064 	},
