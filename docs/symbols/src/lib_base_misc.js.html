  3 /**

 15 /**
 16  * Determines if obj is a boolean
 17  *
 18  * @param {Anything} obj the thing to test if it is a boolean
 19  *
 20  * @returns {Boolean} true if it is a boolean false otherwise
 21  */
 22 comb.isBoolean = function(obj) {
 23     var undef, type = typeof obj;
 24     return obj != undef && type == "boolean" || type == "Boolean";
 25 };

 27 /**
 28  * Determines if obj is undefined
 29  *
 30  * @param {Anything} obj the thing to test if it is undefined
 31  * @returns {Boolean} true if it is undefined false otherwise
 32  */
 33 comb.isUndefined = function(obj) {
 34     var undef;

 72 var isInstance = function(obj, clazz) {
 73     if (typeof clazz == "function") {
 74         return obj instanceof clazz;
 75     } else {
 76         return false;
 77     }
 78 };
