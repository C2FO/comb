function (){
   var ret = undefined, stack, next = this.__next;
   if (next >= 0) {
       stack = this.__stack;
       ret = stack[next];
       stack[this.__next--] = undefined;
   }
   return ret;
           
}

remove Function Public

Removes an element from this stack.

Arguments
  • obj :

    the data to remove.

Returns
  • Boolean

    true if the element was removed, false otherwise.

Source 
function (obj){
   var index = this.__stack.indexOf(obj), ret = false;
   if (index != -1) {
       if (index == this.__next) {
           this.pop();
       } else {
           this.__stack.splice(index, 1);
           this.__next--;
       }
       ret = true;
   }
   return ret;
           
}
