remove Function Public

    the data to remove.

Returns
  • Boolean

Source 
function (obj){
   var index = this.__queue.indexOf(obj), ret = false;
   if (index != -1) {
       if (index == this.__next) {
           this.dequeue();
       } else {
           this.__queue.splice(index, 1);
           this.__last--;
       }
       ret = true;
   }
   return ret;
           
}
